# 引入

import PressActQRCodeDialog from 'press-plus/press-act-q-r-code-dialog/press-act-q-r-code-dialog';

export default {
  components: {
    PressActQRCodeDialog,
  }
}

# 基础用法

<PressActQRCodeDialog
  :show.sync="show"
/>

export default {
  data() {
    return {
      show: true,
    }
  }
}
